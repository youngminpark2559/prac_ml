001-lab. basic of tensorflow, node, session, placeholder, constant import tensorflow as tf # @ # Create a constant operation # This operation will be added as node on default graph hello = tf.constant("Hello, TensorFlow!") # start tensorflow session sess = tf.Session() # run the operation and get result print(sess.run(hello)) # < b'Hello, TensorFlow!' # @ # Let's talk about more complex computational graph # First, we create nodes node1 = tf.constant(3.0, tf.float32) # tf.float32 implicitly node2 = tf.constant(4.0) # We create 'add' node node3 = tf.add(node1, node2) print("node1: ", node1) print("node2: ", node2) print("node3: ", node3) # < node1: Tensor("Const_1:0", shape=(), dtype=float32) # < node2: Tensor("Const_2:0", shape=(), dtype=float32) # < node3: Tensor("Add:0", shape=(), dtype=float32) # We run tensorflow code in this way by using session # We create session obejct sess = tf.Session() # We run session on nodes print("sess.run(node1): ", sess.run(node1)) print("sess.run(node2): ", sess.run(node2)) print("sess.run(node3): ", sess.run(node3)) # < sess.run(node1): 3.0 # < sess.run(node2): 4.0 # < sess.run(node3): 7.0 # We want to implement following scheme # img 7f189664-5ca0-4d6d-8646-bb186b7ed5c4 # For this, we use placeholder node a = tf.placeholder(tf.float32) b = tf.placeholder(tf.float32) # '+' provides shortcut for tf.add(a, b) adder_node = a + b # We pass values into placeholder via feed_dict print(sess.run(adder_node, feed_dict={a: 3, b: 4.5})) print(sess.run(adder_node, feed_dict={a: [1, 3], b: [2, 4]})) # < 7.5 # < [ 3. 7.] add_and_triple = adder_node * 3. print(sess.run(add_and_triple, feed_dict={a: 3, b:4.5})) # < 22.5 # @ # Let's talk about tensor, rank, shape # Rank is dimension of array # Shape of following array is [3,2] # [[1,2], [3,4], [5,6]] # [5,6] => 2 # [1,2], [3,4], [5,6] => 3 # Rank 0 tensor; 'scalar' with shape [] # 3 # Rank 1 tensor; 'vector' with shape [3] # [1. ,2., 3.] # Rank 2 tensor; 'matrix' with shape [2, 3] # [[1., 2., 3.], [4., 5., 6.]] # Rank 3 tensor with shape [2, 1, 3] # [[[1., 2., 3.]], [[7., 8., 9.]]] # [[[1.0, 2.0, 3.0]], [[7.0, 8.0, 9.0]]]
