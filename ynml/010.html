@ # JSON is the method which is used to express object in JavaScript # Actually it's different between the method of expressing object with JSON and the method of expressing object with JavaScript # But it's fair to say it's similar them @ # Data types which can be used in JSON are 6 # 1. numerics : 10, 12 # 1. sequence of characters : "hello" # 1. boolean : true, false # 1. null : null # 1. array : [10, "hello", true, null] # 1. object : # { # "key1" : 10, # "key2" : "hello", # "key3" : true, # "key4" : [12, "hello2"], # "key4" : # { # "key" : "hello3" # } # } @ # The most general formation of JSON is array or object are located outter area because sending simple word, number can be processed by ordinal data form without JSON json_str = """ [ {"name" : "apple", "price" : 1000}, {"name" : "banana", "price" : 2000}, {"name" : "pear", "price" : 3000}, {"name" : "orange", "price" : 4000}, {"name" : "prune", "price" : 5000} ] """ @ import json # If you want to convert external sequence of characters into data type of python # you can you json.load() when you load one value and json.loads() when you load multiple values # output is python data type output = json.loads(json_str) print(output) print(type(output)) # class 'list' print() # If you want to convert data type of python into external sequence of characters(JSON) # you can you json.dump() when you dump one value and json.dumps() when you dump multiple values #
