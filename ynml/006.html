@ # We scraped web data by using built in url library in python # When you do login, we should use "session" technique # But since Dealing with "session" only by url library is pretty difficult, we'd better use external module "requests" @ # pip install requests @ import requests # first, you create session instance session = requests.session() # and you can use get, post, put, delete methods # url is url what we want to scrap # session.get(url) # session.post(url) # session.put(url) # session.delete(url) @ # You get response instance response = session.get(url) # And then, you can use raise_for_status() to actually fire your request response.raise_for_status() # And then you can utilize text in response response.text @ url = "http://google.com" session = requests.session() response = session.get(url) response.raise_for_status() print(response.text) # You can see main page of google # Encoding and other things are automatically processed by "requests library" @ # You can make and use an data instance for "appointing data" rather using querystring url = "http://google.com" data = { "a":"10", "b":"20" } session = requests.session() # You can pass querystring data in get() response = session.get(url, data=data) response.raise_for_status() print(response.text) @ # When you encounter some problem with url library, you can resolve 99% of them by this method # 1.
